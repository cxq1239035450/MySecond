//1. 导入 koa
const Koa = require('koa')
//2. 创建应用对象
const app = new Koa()
//3.创建请求  koa本身没有路由模块
app.use((ctx)=>{

})

# 1.2 ctx上下文介绍

Koa Context 将 Node 的 request 和 response 对象封装到一个对象中，该对象为编写 Web 应用和 API 提供了许多有用的方法。

API作用
ctx.req原先的 request 对象。
ctx.res原先的 response 对象。
ctx.requestKoa处理后的 Request 对象。
ctx.responseKoa处理后的 response 对象。
ctx.throw([status], [msg], [properties])抛出错误
ctx.cookies.get(name, [options])获取cookies
ctx.cookies.set(name, value, [options])设置cookies
ctx.redirect('url')设置重定向的路径

# 1.3 Koa-router路由

const app = new Koa()

# 1.4.3 定义错误处理中间件

一般Koa中的错误分为三类

  • 404: 当请求的资源找不到, 或者没有通过ctx.body返回时, 由koa自动返回
  • 手动抛出: 通过ctx.throw手动抛出
  • 500: 运行时错误
const Koa = require('koa');
const app = new Koa();

// 方法一
//在头部编写一个错误处理的中间件
app.use(async(ctx,next)=>{
    try{
       await next() 
    }catch(err){
        //错误处理
        ctx.status = 500
        ctx.body = err.message
        //ctx.app.emit('error',err,ctx) 将异常抛出
    }
})

// 中间件函数
app.use(async (ctx, next) => {
  JSON.paser('111') //错误代码
  await next(); // 执行下一个中间件
});
app.use((ctx, next) => {
  
});

// 方法二 
// 如果中间件中拦截了报错  该方法就不执行
app.on('error',(err, ctx)=>{
    ctx.status = 500
    ctx.body = err.message
})

# 1.4.4 Koa 中间件机制

koa中间件采用洋葱模型(执行顺序是先进后出) 所以如果中间有异步错误没有等待就拦截不到

async function f1(ctx, next) {
  console.log('1');
  await next();
  console.log('6');
}

async function f2(ctx, next) {
  console.log('2');
  await next();
  console.log('5');
}

async function f3(ctx, next) {
  console.log('3');
  next()
  console.log('4');
}
// 打印结果是 1 2 3 4 5 6

# 1.4.5 与express区别

Express 使用普通的回调函数，一种线性的逻辑，在同一个线程上完成所有的 HTTP 请求

const Express = require('express')
const app = new Express();
const sleep = () => new Promise(resolve => setTimeout(function(){resolve(1)}, 2000))
const port = 3000

function f1(req, res, next) {
  console.log('1');
  next();
  console.log('6');
}

function f2(req, res, next) {
  console.log('2');
  next();
  console.log('5');
}

async function f3(req, res) {
  console.log('3');
  await sleep();
  console.log('4');
  res.send('Hello World!')
}
// 打印结果是  1 2 3 5 6 4  
// express 中间件实现是基于 Callback 回调函数同步的，它不会去等待异步（Promise）完成
